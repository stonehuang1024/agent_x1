一、Tool Call 核心原理
1. 架构本质
Tool Call（函数调用）并非大模型真的"执行"了代码，而是一种结构化生成能力。其核心机制是：
plain
复制
┌─────────────────────────────────────────────────────────┐
│  用户请求 ──► 大模型分析 ──► 生成JSON格式调用指令 ──► 外部执行 ──► 结果返回  │
│         (需要工具?)    (工具名称+参数)      (系统执行)    (注入上下文)   │
└─────────────────────────────────────────────────────────┘
关键洞察：模型输出的是文本，但这段文本被约定为特定格式（通常是JSON），系统解析后执行对应函数，再将结果喂回模型。
2. 与传统Prompt Engineering的区别
表格
方式	原理	可控性	可靠性
Prompt工程	让模型生成代码块，正则提取	低	低（易幻觉）
Tool Call（Native）	模型原生支持JSON Schema约束	高	高（参数校验）
ReAct模式	思维链+行动交替	中	中
现代大模型（GPT-4、Claude、Kimi等）采用Native Tool Calling，在训练阶段就注入了函数调用能力。

阶段2：对话循环（核心机制）
plain
复制
第1轮：
用户: "北京今天天气怎么样?"
模型: ► 分析：需要实时数据 → 生成工具调用
     {
       "tool_calls": [{
         "id": "call_123",
         "type": "function",
         "function": {
           "name": "get_weather",
           "arguments": "{\"location\": \"北京\", \"unit\": \"celsius\"}"
         }
       }]
     }

系统: ► 解析JSON → 执行get_weather("北京", "celsius") 
     ► 获得结果: {"temperature": 22, "condition": "晴"}

第2轮（自动，对用户透明）：
系统: 将工具结果以特殊角色(role=tool)注入上下文
模型: ► 基于真实数据生成最终回答
     "北京今天天气晴朗，气温22摄氏度，适合出行！"

import json
import requests
from typing import List, Dict, Callable, Any
from dataclasses import dataclass, asdict
from enum import Enum

class Role(Enum):
    SYSTEM = "system"
    USER = "user"
    ASSISTANT = "assistant"
    TOOL = "tool"

@dataclass
class Message:
    role: str
    content: str = None
    tool_calls: List[Dict] = None
    tool_call_id: str = None
    name: str = None  # 工具名，用于tool角色
    
    def to_dict(self):
        result = {"role": self.role}
        if self.content:
            result["content"] = self.content
        if self.tool_calls:
            result["tool_calls"] = self.tool_calls
        if self.tool_call_id:
            result["tool_call_id"] = self.tool_call_id
        if self.name:
            result["name"] = self.name
        return result

class Tool:
    """工具包装器"""
    def __init__(self, name: str, description: str, 
                 parameters: Dict, func: Callable):
        self.name = name
        self.description = description
        self.parameters = parameters
        self.func = func
        self.schema = {
            "type": "function",
            "function": {
                "name": name,
                "description": description,
                "parameters": parameters
            }
        }
    
    def execute(self, arguments: str) -> str:
        """执行工具调用"""
        args = json.loads(arguments)
        result = self.func(**args)
        return json.dumps(result, ensure_ascii=False)

class ToolCallEngine:
    """工具调用引擎 - 核心协调器"""
    
    def __init__(self, api_key: str, base_url: str, model: str):
        self.api_key = api_key
        self.base_url = base_url
        self.model = model
        self.tools: Dict[str, Tool] = {}
        self.messages: List[Message] = []
        
    def register_tool(self, tool: Tool):
        """注册工具"""
        self.tools[tool.name] = tool
        
    def chat(self, user_input: str) -> str:
        """主对话循环"""
        # 1. 添加用户消息
        self.messages.append(Message(role=Role.USER.value, content=user_input))
        
        # 2. 第一轮：获取模型响应（可能包含工具调用）
        response = self._call_llm()
        assistant_msg = self._parse_response(response)
        self.messages.append(assistant_msg)
        
        # 3. 检查是否需要工具调用
        if assistant_msg.tool_calls:
            # 4. 执行所有工具调用（并行处理）
            tool_results = self._execute_tools(assistant_msg.tool_calls)
            
            # 5. 将工具结果加入上下文
            for result in tool_results:
                self.messages.append(result)
            
            # 6. 第二轮：让模型基于工具结果生成最终回答
            final_response = self._call_llm()
            final_msg = self._parse_response(final_response)
            self.messages.append(final_msg)
            
            return final_msg.content
        
        return assistant_msg.content
    
    def _call_llm(self) -> Dict:
        """调用大模型API"""
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        
        # 构造工具schema列表
        tools = [tool.schema for tool in self.tools.values()]
        
        payload = {
            "model": self.model,
            "messages": [msg.to_dict() for msg in self.messages],
            "tools": tools if tools else None,
            "tool_choice": "auto" if tools else None
        }
        
        # 移除None值
        payload = {k: v for k, v in payload.items() if v is not None}
        
        response = requests.post(
            f"{self.base_url}/chat/completions",
            headers=headers,
            json=payload,
            timeout=30
        )
        return response.json()
    
    def _parse_response(self, response: Dict) -> Message:
        """解析模型响应"""
        choice = response["choices"][0]
        message = choice["message"]
        
        return Message(
            role=Role.ASSISTANT.value,
            content=message.get("content"),
            tool_calls=message.get("tool_calls")
        )
    
    def _execute_tools(self, tool_calls: List[Dict]) -> List[Message]:
        """执行工具调用并返回结果消息"""
        results = []
        
        for call in tool_calls:
            tool_name = call["function"]["name"]
            arguments = call["function"]["arguments"]
            call_id = call["id"]
            
            print(f"🔧 执行工具: {tool_name}({arguments})")
            
            if tool_name in self.tools:
                tool = self.tools[tool_name]
                try:
                    output = tool.execute(arguments)
                    print(f"✅ 结果: {output}")
                except Exception as e:
                    output = json.dumps({"error": str(e)})
                    print(f"❌ 错误: {e}")
            else:
                output = json.dumps({"error": f"工具 {tool_name} 未找到"})
            
            # 构造tool角色消息（OpenAI格式）
            results.append(Message(
                role=Role.TOOL.value,
                content=output,
                tool_call_id=call_id,
                name=tool_name
            ))
        
        return results

def calculate(expression: str) -> Dict:
    """安全计算表达式"""
    try:
        # 注意：实际生产环境需要更严格的安全沙箱
        allowed_chars = set("0123456789+-*/(). ")
        if not all(c in allowed_chars for c in expression):
            return {"error": "非法字符"}
        result = eval(expression)  # 简化示例，勿用于生产
        return {"expression": expression, "result": result}
    except Exception as e:
        return {"error": str(e)}

import subprocess
import tempfile
import os

def safe_execute(code: str) -> str:
    """在隔离环境中执行代码"""
    with tempfile.NamedTemporaryFile(mode='w', suffix='.py') as f:
        f.write(code)
        f.flush()
        
        # 使用Docker或受限子进程
        result = subprocess.run(
            ['python', '-c', f'exec(open("{f.name}").read())'],
            capture_output=True,
            text=True,
            timeout=5,
            # 限制资源
            preexec_fn=lambda: os.nice(19)
        )
        return result.stdout or result.stderr

class HumanInTheLoop(Tool):
    """高风险操作需要人类确认"""
    def execute(self, arguments: str) -> str:
        args = json.loads(arguments)
        print(f"⚠️  即将执行: {self.name}({args})")
        confirm = input("确认执行? [y/N]: ")
        if confirm.lower() != 'y':
            return json.dumps({"status": "cancelled_by_user"})
        return super().execute(arguments)
六、总结
Tool Call的本质是"结构化生成+系统执行"的协作模式：
模型侧：负责意图识别、参数抽取、结果整合（它擅长的语言理解）
系统侧：负责权限校验、实际执行、安全防护（它擅长的精确控制）
协议层：JSON Schema作为契约，确保双方理解一致
这种模式让大模型从"静态知识库"转变为"动态智能体"，是构建AI Agent的核心基础设施。
